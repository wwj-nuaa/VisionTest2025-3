#git 操作

 

git config --global user.name "xxx"  配置用户名
git config --global user.emal "xxx"  配置邮件

git config list 查看配置

git init  初始化新仓库

git clone /a/b/github.com       复制代码到本地

git add remote origin 本地仓库地址
git add remote upstream 原仓库地址

git status  查看当前状态

git add 文件名  添加指定文件
git add .   添加所有修改

git commit -m "提交说明"

git log 显示日志
git log -n 显示n行日志
git log --oneline  简洁版日志

分支操作：
    git branch 分支名  创建分支
    git checkout 分支名  切换分支
    git checkout -b 分支名  创建并切换分支
    git branch -a  查看所有分支
    git branch -d 分支名  删除分支


获取更新：
    git fetch origin  获取不合并
    git pull  获取且合并
    git pull --rebase  获取且变基

提交时出现冲突：
    git add .
    git rebase --continue   继续变基
    git rebase --abort  放弃变基
    git rebase --skip  放弃冲突提交

git证书出现问题(更新系统证书)：
    sudo apt update
    sudo apt install --reinstall ca-certificates
    sudo update-ca-certificates --fresh
